import "./herosection.css";
import { useState, useEffect } from "react";

const slides = [
        {id: 1, image: "/src/assets/FarmerImg.jpg", spanText: "Farmer.", },
        {id: 1, image: "/src/assets/ShopOwnerImg.jpg", spanText: "Shop Owner.", },
        {id: 1, image: "/src/assets/ArtisanImg.jpg", spanText: "Artisan.", },
        {id: 1, image: "/src/assets/NGOImg.jpg", spanText: "NGO.", },
    ];

function Herosection(){
    
    const total = slides.length;
    const [current, setCurrent] = useState(0);
    const [next, setNext] = useState(1);
    const [transitioning, setTransitioning] = useState(false);
 
    useEffect(() => {
    const interval = setInterval(() => {
      const nextIdx = (current + 1) % total;
      setNext(nextIdx);
      setTransitioning(true); // start expanding the right slide
 
      setTimeout(() => {
        setCurrent(nextIdx); // commit the new current
        setTransitioning(false); // reset
      }, 600); // match CSS transition duration
 
    }, 3000);
 
    return () => clearInterval(interval);
  }, [current, total]);
 
    const prevIndex = (current - 1 + total) % total;

    const leftClass  = "slide side";
    const centerClass = transitioning ? "slide side"   : "slide active";
    const rightClass  = transitioning ? "slide active" : "slide side";
    
    const centerImage = slides[current].image;
    const rightImage  = transitioning ? slides[next].image : slides[(current + 1) % total].image;

    return(
        <div className="heroSection">
            <div className="textAndButtons">
                <h1>eNaira Offline Access for Every Nigerian{" "} <span key={current} className="dynamicSpan">{slides[current].spanText}</span></h1>
            </div>
            <div className="slider">
                <div className="sliderTrack">
                <div className={leftClass}>
                    <img src={slides[prevIndex].image} alt="" />
                </div>
                <div className={centerClass}>
                    <img src={centerImage} alt="" />
                </div>
                <div className={rightClass}>
                    <img src={rightImage} alt="" />
                </div>
                </div>
            </div>
        </div>
    )
}
export default Herosection;
